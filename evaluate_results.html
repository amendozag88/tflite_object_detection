import matplotlib.pyplot as plt
from sklearn.metrics import confusion_matrix, classification_report

def parse_result_file(path, no_det_label="no_detections"):
    best_pred = {}

    with open(path, "r") as f:
        for line in f:
            line = line.strip()
            if not line or line.lower().startswith("image_path"):
                continue

            parts = [p.strip() for p in line.split(",")]
            img_path = parts[0]
            true_label = img_path.split("/")[0].lower()

            if parts[1] == no_det_label:
                pred_label = no_det_label
                conf = -1.0
            else:
                pred_label = parts[1].lower()
                try:
                    conf = float(parts[2])
                except:
                    conf = 0.0

            if img_path not in best_pred or conf > best_pred[img_path][0]:
                best_pred[img_path] = (conf, true_label, pred_label)

    y_true = [v[1] for v in best_pred.values()]
    y_pred = [v[2] for v in best_pred.values()]
    return y_true, y_pred


def summarize_for_notebook(path, valid_classes, model_name=[], no_det_label="no_detections"):
    valid_classes = [c.lower() for c in valid_classes]

    model_name = model_name.split("_")[1:]

    raw_true, raw_pred = parse_result_file(path, no_det_label=no_det_label)

    y_true, y_pred = [], []
    for t, p in zip(raw_true, raw_pred):
        if t in valid_classes:
            y_true.append(t)
            y_pred.append(p if p in valid_classes else "other")

    labels = valid_classes + ["other"]

    # Confusion matrix (counts)
    cm = confusion_matrix(y_true, y_pred, labels=labels)
    acc = np.mean(np.array(y_true) == np.array(y_pred))
    print(f"Overall accuracy: {acc:.3f}\n")
    # === Raw counts heatmap ===
    plt.figure(figsize=(8, 6))
    sns.heatmap(cm, annot=True, fmt="d", cmap="Greens",
                xticklabels=labels, yticklabels=labels)
    plt.title(f"Confusion Matrix - {model_name}")
    plt.xlabel("Predicted")
    plt.ylabel("True")
    plt.show()

    # === Classification report table ===
    print("\nClassification Report:\n")
    class_report = classification_report(y_true, y_pred, labels=labels, zero_division=0)
    print(class_report)

    return cm, y_true, y_pred, class_report


def summarize_results(file_path, valid_classes=['toaster','oven','truck','cow','airplane','train'], no_det_label="no_detections"):
    """
    Summarize results **restricted** to a set of valid GT classes.
    """
    print("=" * 80)
    print(f"File: {file_path}")
    print("-" * 80)

    raw_true, raw_pred = parse_result_file(file_path, no_det_label=no_det_label)

    valid_classes = [c.lower() for c in valid_classes]

    # Restrict to samples whose TRUE label is in valid classes
    y_true = []
    y_pred = []
    for t, p in zip(raw_true, raw_pred):
        if t in valid_classes:
            y_true.append(t)
            # If predicted label is outside valid set, map it to "other"
            y_pred.append(p if p in valid_classes else "other")

    if not y_true:
        print("No valid samples found.")
        return

    labels = valid_classes + ["other"]  # include "other" column
    cm = confusion_matrix(y_true, y_pred, labels=labels)
    acc = np.mean(np.array(y_true) == np.array(y_pred))

    class_counts = Counter(y_true)

    print(f"Valid ground-truth classes: {valid_classes}")
    print(f"Total valid images: {len(y_true)}")
    print(f"Overall accuracy: {acc:.3f}\n")

    print("Samples per GT class:")
    for c in valid_classes:
        print(f"  {c:15s}: {class_counts[c]}")
    print()

    # Print confusion matrix
    print("Labels order in confusion matrix:", labels)
    header = "true\\pred".ljust(15) + " " + " ".join(f"{lbl:15s}" for lbl in labels)
    print(header)
    for i, true_lbl in enumerate(labels):
        row_vals = " ".join(f"{cm[i, j]:15d}" for j in range(len(labels)))
        print(f"{true_lbl:15s} {row_vals}")
    print()

    print("Classification report:")
    print(classification_report(y_true, y_pred, labels=labels, zero_division=0))
